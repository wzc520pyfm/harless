# harless v0.1 Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Ship `harless` v0.1 — a single-package npm CLI (≤ 500 LOC) that injects 12 SKILL files + 5 shell scripts + AGENTS.md routing into a user repo, gated by smoke tests on Claude Code, Cursor, and Codex CLI.

**Architecture:** TypeScript CLI with `commander`-style entry, hand-rolled string templating (no engine), `@inquirer/prompts` for interaction, vitest for tests. Templates live as plain files under `packages/cli/src/templates/` and ship 1:1 to user `.agents/`. Hash-based per-file update strategy. Zero runtime, zero daemon, zero global config writes.

**Tech Stack:** Node 18+, TypeScript 5.x, vitest, execa, @inquirer/prompts, picocolors, GitHub Actions, VitePress (docs).

**Spec:** `[docs/superpowers/specs/2026-04-20-harless-design.md](../specs/2026-04-20-harless-design.md)` — authoritative source. All file paths, behaviors, exit codes, LOC budgets, and DoD criteria come from there.

#!/usr/bin/env bash

echo "# Pre-Implementation Verification — $(date -Iseconds)" > "$OUT"

automated() {
  local id="$1" desc="$2" cmd="$3"
  echo "## $id — $desc" >> "$OUT"
  echo '```' >> "$OUT"
  if eval "$cmd" >> "$OUT" 2>&1; then echo "✓ PASS" >> "$OUT"; else echo "✗ FAIL"; fi
  echo '```' >> "$OUT"

  local id="$1" desc="$2" steps="$3"

  echo "$steps" >> "$OUT"
  read -rp "$id passed? [y/N/u]: " ans
  case "$ans" in y|Y) echo "Result: ✓ PASS" >> "$OUT";;
                 u|U) echo "Result: ? UNKNOWN" >> "$OUT";;
                 *)   echo "Result: ✗ FAIL" >> "$OUT";; esac
}

```ts
export type UpdateAction = "keep" | "overwrite" | "conflict" | "recreate";

export interface UpdateInputs {
  diskExists: boolean;
  diskHash: string | null;
  oldTemplateHash: string;
  newTemplateHash: string;
}

export function decideUpdateAction(i: UpdateInputs): UpdateAction {
  if (!i.diskExists) return "recreate";
  const diskPristine = i.diskHash === i.oldTemplateHash;
  const templateChanged = i.oldTemplateHash !== i.newTemplateHash;
  if (diskPristine && templateChanged) return "overwrite";
  if (!diskPristine && !templateChanged) return "keep";
  if (diskPristine && !templateChanged) return "keep";
  return "conflict";
}
```

```ts
import fs from "node:fs";
import path from "node:path";

export type Framework = "next" | "vite" | "remix" | "astro" | "svelte" | "react" | "unknown";
export type PackageManager = "pnpm" | "yarn" | "npm" | "bun" | "unknown";
export type StackKind = "web" | "generic";

export interface StackInfo {
  framework: Framework;
  packageManager: PackageManager;
  kind: StackKind;
  workspace: boolean;
}

const FRAMEWORK_KEYS: Array<[Framework, string[]]> = [
  ["next", ["next"]],
  ["remix", ["@remix-run/node", "@remix-run/react"]],
  ["astro", ["astro"]],
  ["svelte", ["@sveltejs/kit", "svelte"]],
  ["vite", ["vite"]],
  ["react", ["react"]]
];

export function detectStack(cwd: string): StackInfo {
  const pkgPath = path.join(cwd, "package.json");
  let deps: Record<string, string> = {};
  if (fs.existsSync(pkgPath)) {
    const pkg = JSON.parse(fs.readFileSync(pkgPath, "utf8"));
    deps = { ...(pkg.dependencies ?? {}), ...(pkg.devDependencies ?? {}) };
  }
  let framework: Framework = "unknown";
  for (const [name, keys] of FRAMEWORK_KEYS) {
    if (keys.some((k) => k in deps)) { framework = name; break; }
  }
  const packageManager: PackageManager =
    fs.existsSync(path.join(cwd, "pnpm-lock.yaml")) ? "pnpm" :
    fs.existsSync(path.join(cwd, "yarn.lock")) ? "yarn" :
    fs.existsSync(path.join(cwd, "bun.lockb")) ? "bun" :
    fs.existsSync(path.join(cwd, "package-lock.json")) ? "npm" : "unknown";
  const workspace =
    fs.existsSync(path.join(cwd, "pnpm-workspace.yaml")) ||
    fs.existsSync(path.join(cwd, "lerna.json")) ||
    (fs.existsSync(pkgPath) && Array.isArray(JSON.parse(fs.readFileSync(pkgPath, "utf8")).workspaces));
  const kind: StackKind = framework === "unknown" ? "generic" : "web";
  return { framework, packageManager, kind, workspace };
}
```

```ts
export interface McpJson { mcpServers: Record<string, unknown>; [k: string]: unknown }

export function mergeMcpJson(
  existing: Partial<McpJson> | null,
  toAdd: Record<string, unknown>
): McpJson {
```

```ts
const BEGIN = "<!-- BEGIN harless v0.1 -->";
const END = "<!-- END harless v0.1 -->";

export function mergeAgentsMd(existing: string | null, blockBody: string): string {
  const block = `${BEGIN}\n${blockBody.trim()}\n${END}`;
  if (existing == null) return block + "\n";
  if (existing.includes(BEGIN) && existing.includes(END)) {
    const re = new RegExp(`${escapeRe(BEGIN)}[\\s\\S]*?${escapeRe(END)}`);
    return existing.replace(re, block);
  }
  const sep = existing.endsWith("\n") ? "\n" : "\n\n";
  return existing + sep + block + "\n";
}

function escapeRe(s: string): string {
  return s.replace(/[.*+?^${}()|[\]\\]/g, "\\$&");
```

```ts
export interface HarlessConfig {
  version: string;
  installedAt: string;
  stack: { detected: string; packageManager: string };
  agents: string[];
  defaultAgentCmd: string;
  modules: Record<string, ModuleConfig>;
}
export interface ModuleConfig {
  enabled: boolean;
  files?: Array<{ path: string; hash: string }>;
  mcp?: string[];
}

export function parseConfig(raw: string): HarlessConfig {
  const o = JSON.parse(raw);
  for (const k of ["version", "installedAt", "defaultAgentCmd"] as const) {
    if (typeof o[k] !== "string") throw new Error(`config: missing or non-string ${k}`);
  }
  if (typeof o.stack !== "object" || o.stack == null) throw new Error("config: missing stack");
  if (!Array.isArray(o.agents)) throw new Error("config: agents must be array");
  if (typeof o.modules !== "object" || o.modules == null) throw new Error("config: missing modules");
  for (const [k, v] of Object.entries(o.modules as Record<string, any>)) {
    if (typeof v?.enabled !== "boolean") throw new Error(`config: modules.${k}.enabled must be boolean`);
    if (v.files != null && !Array.isArray(v.files)) throw new Error(`config: modules.${k}.files must be array`);
  }
  return o as HarlessConfig;
}

export function stringifyConfig(c: HarlessConfig): string {
  return JSON.stringify(c, null, 2) + "\n";
}
```

```ts
import os from "node:os";
import path from "node:path";

const CACHE = path.join(os.homedir(), ".cache", "harless", "version-check.json");
const TTL_MS = 24 * 60 * 60 * 1000;

export async function latestVersion(opts: { offline: boolean; now?: number; fetchImpl?: typeof fetch }): Promise<string | null> {
  if (opts.offline) return null;
  const now = opts.now ?? Date.now();
  if (fs.existsSync(CACHE)) {
    try {
      const c = JSON.parse(fs.readFileSync(CACHE, "utf8"));
      if (now - c.checkedAt < TTL_MS) return c.version ?? null;
    } catch { /* ignore */ }
  }
  const f = opts.fetchImpl ?? fetch;
  const ctrl = new AbortController();
  const t = setTimeout(() => ctrl.abort(), 1000);
  try {
    const res = await f("https://registry.npmjs.org/harless/latest", { signal: ctrl.signal });
    if (!res.ok) return null;
    const j: any = await res.json();
    fs.mkdirSync(path.dirname(CACHE), { recursive: true });
    fs.writeFileSync(CACHE, JSON.stringify({ checkedAt: now, version: j.version }));
    return j.version;
  } catch {
    return null;
  } finally {
    clearTimeout(t);
  }
}
```

```ts
export async function run(argv: string[]): Promise<void> {
  const [cmd, ...rest] = argv;
  const flags = parseGlobalFlags(rest);
  switch (cmd) {
    case "init":    return initCmd(flags);
    case "add":     return addCmd(flags.positional[0], flags);
    case "remove":  return removeCmd(flags.positional[0], flags);
    case "update":  return updateCmd(flags);
    case "doctor":  return doctorCmd(flags);

    }
    case undefined: case "--help": case "-h":

    default:

  }
}

function parseGlobalFlags(args: string[]) {
  const positional: string[] = [];
  const flags: Record<string, string | boolean> = { yes: false, offline: false };
  for (const a of args) {

    else if (a === "--offline") flags.offline = true;
    else if (a.startsWith("--")) {

      flags[k] = v;
    } else positional.push(a);
  }
  return { ...flags, positional } as any;
}
```

```ts
    fs.mkdirSync(path.dirname(dst), { recursive: true });
    // Fresh install: copy. Repair (4.5b): caller deletes first, so dst is missing.
    // `add` on already-installed module: caller short-circuits before reaching here.
    if (fs.existsSync(dst)) throw new Error(`installer: refusing to overwrite ${dst} (use repair or add idempotency check)`);
    fs.copyFileSync(src, dst);
    if (rel.endsWith(".sh")) fs.chmodSync(dst, 0o755);
    files.push({ path: `.agents/${rel}`, hash: sha256(fs.readFileSync(dst)) });
  }
  if (m.mcp) {
    const mcpPath = path.join(cwd, ".mcp.json");
    const existing = fs.existsSync(mcpPath) ? JSON.parse(fs.readFileSync(mcpPath, "utf8")) : null;
    fs.writeFileSync(mcpPath, JSON.stringify(mergeMcpJson(existing, m.mcp), null, 2) + "\n");
  }
  config.modules[module] = { enabled: true, files, ...(m.mcp ? { mcp: Object.keys(m.mcp) } : {}) };
  return config.modules[module];
}

export function rewriteAgentsMd(cwd: string, blockBody: string) {
  const p = path.join(cwd, "AGENTS.md");
  const existing = fs.existsSync(p) ? fs.readFileSync(p, "utf8") : null;
  fs.writeFileSync(p, mergeAgentsMd(existing, blockBody));
}

export function distTemplateRoot(): string {
  // Resolve relative to the compiled dist/cli.js location
  const here = new URL("../templates", import.meta.url);
  return here.pathname;
}
```

- **Step 1: Extend `mergeAgentsMd` with row removal**

In the harless block we ship a Markdown table whose rows look like

Per spec §5.3, AGENTS.md rows look like
`| Have an idea, before designing | \`.agents/skills/brainstorming/SKILL.md |`. The first cell is human prose, so we must key off the **path cell** (`.agents//...`). For the` skills` bundle we iterate the 5 discipline paths.

```ts
// added to merge-agents-md.ts
const SKILLS_BUNDLE_PATHS = [
  "skills/brainstorming",
  "skills/tdd",
  "skills/systematic-debugging",
  "skills/verification-before-completion",
  "skills/receiving-review",
];

export function removeRowsFromAgentsBlock(existing: string, moduleName: string): string {
  if (!existing.includes(BEGIN) || !existing.includes(END)) return existing;
  const targets = moduleName === "skills" ? SKILLS_BUNDLE_PATHS : [`${moduleName}/`];
  const beginIdx = existing.indexOf(BEGIN);
  const endIdx   = existing.indexOf(END);
  let block = existing.slice(beginIdx, endIdx);
  for (const t of targets) {
    const re = new RegExp(`^.*\`\\.agents/${escapeRe(t)}[^\`]*\`.*\\n`, "gm");
    block = block.replace(re, "");
  }
  return existing.slice(0, beginIdx) + block + existing.slice(endIdx);
}
```
